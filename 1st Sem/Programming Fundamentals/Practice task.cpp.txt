int main() {

	int arr[3][3];
	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) {
			cout << "Enter the number : ";
			cin >> arr[i][j];
		}
	}
	for (int i = 0, j = 0; j < 2; j++) {
		if (arr[i][j] > arr[i][j + 1]) {
			int temp = arr[i][j+1];
			arr[i][j + 1] = arr[i][j];
			arr[i][j] = temp;
		}
	}
}
